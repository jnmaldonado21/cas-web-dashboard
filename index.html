<script>

  var DATA_URL = 'https://jnmaldonado21.github.io/cas-web-dashboard/data.json';


  document.addEventListener('DOMContentLoaded', function() {

    fetch(DATA_URL + '?t=' + Date.now())

      .then(function(r) {

        if (!r.ok) throw new Error('HTTP ' + r.status);

        return r.json();

      })

      .then(function(data) {

        populate(data);

        document.getElementById('loading').classList.add('hidden');

        setTimeout(function() { document.getElementById('loading').style.display = 'none'; }, 500);

      })

      .catch(function(err) {

        console.error('Data load error:', err);

        document.getElementById('loading').classList.add('hidden');

        setTimeout(function() { document.getElementById('loading').style.display = 'none'; }, 500);

      });

  });


  function populate(data) {

    if (data.reportPeriod) {

      set('reportDate', '90-day report · ' + data.reportPeriod.label);

    }


    // Story 1 — GA4

    if (data.ga4) {

      var g = data.ga4;

      set('s1-sessions-inline', g.sessions.value + ' people');

      set('s1-sessions', g.sessions.value);

      set('s1-newusers', g.newUsers.value);

      set('s1-engage',   g.engagementRate.value);

      set('s1-duration', g.avgDuration.value);

      set('s3-newusers-inline', g.newUsers.value);

      set('f1', g.sessions.value + ' visits');

      set('f2', g.newUsers.value);

      set('f4', g.admissionsVisits.value);

      set('s3-admissions', g.admissionsVisits.value);

      buildSparkline(g.monthlyTrend);

      buildTopPages(g.topPages);

      buildDeptBreakdown(g.deptBreakdown);

    }


    // Story 2 — Siteimprove

    if (data.siteimprove) {

      var s = data.siteimprove;

      set('s2-access',     s.accessibilityScore.value);

      set('s2-seo',        s.seoScore.value);

      set('s2-broken',     s.brokenLinks.value);

      set('s2-issues',     s.qualityIssues.value);

      set('s2-access-val', s.accessibilityScore.value);

      set('s2-seo-val',    s.seoScore.value);

      document.getElementById('s2-access-bar').style.width = (s.accessibilityScore.raw || 0) + '%';

      document.getElementById('s2-seo-bar').style.width    = (s.seoScore.raw || 0) + '%';

    }


    // Story 4 — Freshdesk

    if (data.freshdesk) {

      var f = data.freshdesk;

      set('s4-tickets',    f.ticketVolume.value);

      set('s4-response',   f.avgResponseTime.value);

      set('s4-csat',       f.csatScore.value);

      set('s4-resolution', f.resolutionRate.value);

    }


    // Story 5 — Centers inventory

    buildFootprintTable(data.centersInventory);

  }


  function buildSparkline(trend) {

    var el = document.getElementById('sparkline');

    if (!trend || !trend.values) return;

    var max = Math.max.apply(null, trend.values) || 1;

    var shades = ['#d1a0a0','#c07070','#a85050','#8c3535','#6b1f1f','#500000'];

    el.innerHTML = trend.values.map(function(v, i) {

      var pct = Math.max(8, Math.round((v / max) * 100));

      return '<div class="spark-col">' +

        '<div class="spark-bar" style="height:' + pct + '%;background:' + shades[Math.min(i, 5)] + '"></div>' +

        '<div class="spark-mo">' + (trend.labels[i] || '') + '</div></div>';

    }).join('');

  }


  function buildTopPages(pages) {

    var el = document.getElementById('topPages');

    if (!pages || !pages.length) return;

    var max = pages[0].views || 1;

    el.innerHTML = pages.map(function(p) {

      var pct = Math.round((p.views / max) * 100);

      var lbl = p.path.length > 16 ? p.path.substring(0, 15) + '\u2026' : p.path;

      var v   = p.views >= 1000 ? (p.views / 1000).toFixed(1) + 'k' : p.views;

      return '<div class="page-row">' +

        '<div class="page-path" title="' + p.path + '">' + lbl + '</div>' +

        '<div class="page-track"><div class="page-fill" style="width:' + pct + '%"></div></div>' +

        '<div class="page-views">' + v + '</div></div>';

    }).join('');

  }


  function buildDeptBreakdown(depts) {

    var el = document.getElementById('deptBreakdown');

    if (!depts || !depts.length) {

      el.innerHTML = '<div style="font-size:12px;color:var(--gray-500);grid-column:1/-1">Department breakdown not available — check GA4 property configuration.</div>';

      return;

    }

    var max = depts[0].sessions || 1;

    el.innerHTML = depts.slice(0, 18).map(function(d) {

      var pct = Math.round((d.sessions / max) * 100);

      var v   = d.sessions >= 1000 ? (d.sessions / 1000).toFixed(1) + 'k' : d.sessions;

      return '<div class="dept-item">' +

        '<div class="dept-name" title="' + d.name + '">' + d.name + '</div>' +

        '<div class="dept-track"><div class="dept-fill" style="width:' + pct + '%"></div></div>' +

        '<div class="dept-val">' + v + '</div></div>';

    }).join('');

  }


  function buildFootprintTable(inventory) {

    var el = document.getElementById('fpBody');

    if (!inventory) return;

    var monitored = inventory.filter(function(s) { return s.monitored; }).length;

    var external  = inventory.filter(function(s) { return !s.monitored; }).length;

    set('fp-monitored', String(monitored + 19)); // 19 cascade dept sites + monitored centers

    set('fp-external',  String(external));

    set('fp-total',     String(monitored + external + 19));


    el.innerHTML = inventory.map(function(s) {

      var dot = s.monitored

        ? '<span class="dot-green"></span>In scope'

        : '<span class="dot-gray"></span>External';

      return '<tr>' +

        '<td><a href="' + s.url + '" target="_blank" rel="noopener" style="color:var(--maroon-primary);text-decoration:none">' + s.name + '</a></td>' +

        '<td><span class="platform-badge">' + s.platform + '</span></td>' +

        '<td>' + dot + '</td></tr>';

    }).join('');

  }


  function set(id, val) {

    var el = document.getElementById(id);

    if (el) el.textContent = val;

  }

</script>
